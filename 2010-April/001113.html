On debian/ubuntu: (# lines are comments...)

pavel at gandalf:~$ sudo bash
[sudo] password for pavel:
# gain access to run the rigth command as root... 
root at gandalf:~# dpkg-reconfigure tzdata
# here you have a ncurses interface to select your time zone
Current default time zone: 'America/Havana'
Local time is now:      Wed Apr 14 13:22:26 CDT 2010.
Universal Time is now:  Wed Apr 14 17:22:26 UTC 2010.
# ok, set the date time as local time
root at gandalf:~# date -s "Wed Apr 14 13:26:39 CDT 2010"
# chech your correct utc time
root at gandalf:~# date -u
wed apr 14 17:27:28 UTC 2010
# store your date to the cmos, and exit
root at gandalf:~# hwclock --systohc
root at gandalf:~# exit
